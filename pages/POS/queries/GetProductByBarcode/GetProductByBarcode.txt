SELECT
    p.id AS productId,
    p.code AS productCode,
    p.barcode AS barcode,
    p.name AS productName,
    COALESCE(NULLIF(p.description, ''), p.name) AS description,
    p.sale_price AS unitPrice,
    p.tax_rate_id AS taxRateId,
    COALESCE(tr.rate, 0) AS taxRate,
    p.unit_id AS unitId,
    COALESCE(u.short_name, '') AS unitName,
    p.product_type AS productType,
    p.track_stock AS trackStock,
    COALESCE((
        SELECT SUM(ib.quantity_balance)
        FROM v_inventory_balance ib
        WHERE ib.company_id = p.company_id
          AND ib.product_id = p.id
          AND (
              {{ appsmith.store.warehouseId || 0 }} = 0
              OR ib.warehouse_id = {{ appsmith.store.warehouseId || 0 }}
          )
    ), 0) AS availableStock
FROM products p
LEFT JOIN tax_rates tr ON tr.id = p.tax_rate_id
LEFT JOIN units_of_measure u ON u.id = p.unit_id
WHERE p.is_active = 1
 
  AND (
      TRIM(COALESCE(p.code, '')) = TRIM('{{ BarcodeInput.text || "" }}')
      OR TRIM(COALESCE(p.barcode, '')) = TRIM('{{ BarcodeInput.text || "" }}')
  )
LIMIT 1;
