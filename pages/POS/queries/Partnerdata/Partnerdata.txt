SELECT
    bp.id,
    bp.company_id,
    bp.partner_type,
    bp.code,
    bp.name,
    bp.legal_name,
    bp.contact_person,
    bp.tax_number,
    bp.registration_number,
    bp.email,
    bp.phone,
    bp.mobile,
    bp.address_line1,
    bp.address_line2,
    bp.city_name,
    bp.postal_code,
    bp.payment_term_days,
    bp.credit_limit,
    bp.note,

    pba.bank_account_id,
    ba.account_name,
    ba.account_number,
    ba.iban AS partner_iban,
    ba.swift_code AS partner_swift,
    ba.currency_code AS partner_bank_currency

FROM business_partners bp
LEFT JOIN partner_bank_accounts pba
    ON pba.partner_id = bp.id
   AND pba.is_default = 1
LEFT JOIN bank_accounts ba
    ON ba.id = pba.bank_account_id
WHERE bp.id = {{ SelectPartner.selectedOptionValue }}
  AND bp.company_id = {{ appsmith.store.companyId }}
LIMIT 1;
