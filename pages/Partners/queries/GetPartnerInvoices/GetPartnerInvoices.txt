SELECT
    d.id AS `Document ID`,
    d.company_id AS `Company ID`,
    d.partner_id AS `Partner ID`,
    bp.code AS `Partner Code`,
    bp.name AS `Partner Name`,
    d.document_type AS `Document Type`,
    d.document_number AS `Document Number`,
    d.document_date AS `Document Date`,
    d.due_date AS `Due Date`,
    d.status AS `Status`,
    d.posting_status AS `Posting Status`,
    d.payment_method AS `Payment Method`,
    d.currency_code AS `Currency`,
    d.subtotal_amount AS `Subtotal`,
    d.tax_amount AS `Tax`,
    d.total_amount AS `Total`,
    d.paid_amount AS `Paid`,
    (d.total_amount - d.paid_amount) AS `Open Amount`,
    CASE
        WHEN d.due_date < CURDATE()
         AND (d.total_amount - d.paid_amount) > 0
        THEN 'YES'
        ELSE 'NO'
    END AS `Overdue`
FROM documents d
JOIN business_partners bp ON bp.id = d.partner_id
WHERE d.company_id = {{appsmith.store.companyId}}
  AND bp.company_id = {{appsmith.store.companyId}}
  AND (
      bp.code = {{appsmith.store.selectedPartnerCode || ""}}
      OR bp.name LIKE CONCAT('%', {{appsmith.store.selectedPartnerName || ""}}, '%')
      OR bp.legal_name LIKE CONCAT('%', {{appsmith.store.selectedPartnerName || ""}}, '%')
  )
ORDER BY d.document_date DESC, d.id DESC;
