SELECT
    bp.id AS `Partner ID`,
    bp.code AS `Partner Code`,
    bp.name AS `Partner Name`,
    bp.legal_name AS `Legal Name`,
    bp.contact_person AS `Contact Person`,
    bp.tax_number AS `OIB/VAT`,
    bp.registration_number AS `Registration Number`,
    bp.email AS `Email`,
    bp.phone AS `Phone`,
    bp.mobile AS `Mobile`,
    bp.address_line1 AS `Address`,
    bp.city_name AS `City`,
    bp.postal_code AS `Postal Code`,
    c.code AS `Country Code`,
    c.name AS `Country`,
    bp.payment_term_days AS `Payment Term Days`,
    bp.credit_limit AS `Credit Limit`,
    bp.is_active AS `Active`
FROM business_partners bp
LEFT JOIN countries c ON c.id = bp.country_id
WHERE bp.company_id = {{appsmith.store.companyId}}
  AND (
      {{PartnerSearchInput.text || ''}} = ''
      OR bp.tax_number LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
      OR bp.name LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
      OR bp.legal_name LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
      OR bp.code LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
      OR bp.email LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
      OR bp.phone LIKE CONCAT('%', {{PartnerSearchInput.text}}, '%')
  )
ORDER BY bp.name
LIMIT 200;
