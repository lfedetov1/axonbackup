UPDATE products
SET
  category_id = {{ ProductCategorySelect.selectedOptionValue || null }},
  unit_id = {{ ProductUnitSelect.selectedOptionValue }},
  tax_rate_id = {{ ProductTaxRateSelect.selectedOptionValue || null }},
  code = {{ ProductCodeInput.text }},
  name = {{ ProductNameInput.text }},
  short_name = {{ ProductShortNameInput.text || null }},
  description = {{ ProductDescriptionInput.text || null }},
  barcode = {{ ProductBarcodeInput.text || null }},
  sku = {{ ProductSkuInput.text || null }},
  product_type = {{ ProductTypeSelect.selectedOptionValue || 'GOODS' }},
  track_stock = {{ ProductTrackStockSwitch.isSwitchedOn ? 1 : 0 }},
  is_used_item = {{ ProductUsedItemSwitch.isSwitchedOn ? 1 : 0 }},
  serial_number_required = {{ ProductSerialRequiredSwitch.isSwitchedOn ? 1 : 0 }},
  minimum_stock = {{ ProductMinimumStockInput.text || 0 }},
  maximum_stock = {{ ProductMaximumStockInput.text || null }},
  reorder_level = {{ ProductReorderLevelInput.text || null }},
  purchase_price = {{ ProductPurchasePriceInput.text || 0 }},
  sale_price = {{ ProductSalePriceInput.text || 0 }},
  weight_kg = {{ ProductWeightKgInput.text || null }},
  note = {{ ProductNoteInput.text || null }},
  is_active = {{ ProductActiveSwitch.isSwitchedOn ? 1 : 0 }}
WHERE id = {{ appsmith.store.currentProductId }}
  AND company_id = {{ appsmith.store.companyId }};
