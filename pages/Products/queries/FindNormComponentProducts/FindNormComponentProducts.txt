SELECT
    p.id AS componentProductId,
    p.code AS componentCode,
    p.name AS componentName,
    p.barcode AS componentBarcode,
    p.sku AS componentSku,
    p.unit_id AS unitId,
    u.code AS unitCode,
    u.name AS unitName,
    p.purchase_price AS purchasePrice,
    p.sale_price AS salePrice,
    p.track_stock AS trackStock
FROM products p
LEFT JOIN units_of_measure u ON u.id = p.unit_id
WHERE TRIM({{roductNormComponentSearchInput.text || ""}}) <> ""
  AND p.is_active = 1
  AND (
      LOWER(COALESCE(p.code, "")) LIKE CONCAT("%", LOWER(TRIM({{roductNormComponentSearchInput.text || ""}})), "%")
      OR LOWER(COALESCE(p.name, "")) LIKE CONCAT("%", LOWER(TRIM({{roductNormComponentSearchInput.text || ""}})), "%")
      OR LOWER(COALESCE(p.barcode, "")) LIKE CONCAT("%", LOWER(TRIM({{roductNormComponentSearchInput.text || ""}})), "%")
      OR LOWER(COALESCE(p.sku, "")) LIKE CONCAT("%", LOWER(TRIM({{roductNormComponentSearchInput.text || ""}})), "%")
  )
ORDER BY p.name
LIMIT 30;
