SELECT
    DATE_FORMAT(d.document_date, '%Y-%m') AS `Sort Period`,
    DATE_FORMAT(d.document_date, '%M %Y') AS `Period`,

    COALESCE(SUM(
        CASE
            WHEN d.document_type IN ('SALES_INVOICE', 'POS_SALE')
            THEN di.line_total
            ELSE 0
        END
    ), 0) AS `Sales Total`,

    COALESCE(SUM(
        CASE
            WHEN d.document_type IN ('PURCHASE_RECEIPT', 'PURCHASE_INVOICE')
            THEN di.line_total
            ELSE 0
        END
    ), 0) AS `Purchase Total`,

    COALESCE(SUM(
        CASE
            WHEN d.document_type IN ('SALES_INVOICE', 'POS_SALE')
            THEN di.quantity
            ELSE 0
        END
    ), 0) AS `Sales Quantity`,

    COALESCE(SUM(
        CASE
            WHEN d.document_type IN ('PURCHASE_RECEIPT', 'PURCHASE_INVOICE')
            THEN di.quantity
            ELSE 0
        END
    ), 0) AS `Purchase Quantity`
FROM document_items di
JOIN documents d ON d.id = di.document_id
WHERE di.product_id = {{ProductsTable.selectedRow?.["Product ID"] || ProductsTable.selectedRow?.id || 0}}
  AND d.document_type IN ('SALES_INVOICE', 'POS_SALE', 'PURCHASE_RECEIPT', 'PURCHASE_INVOICE')
  AND d.status <> 'CANCELLED'
GROUP BY
    `Sort Period`,
    `Period`
ORDER BY `Sort Period`;
