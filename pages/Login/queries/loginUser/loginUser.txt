SELECT
  u.id AS user_id,
  u.username,
  u.email,
  u.first_name,
  u.last_name,
  c.id AS company_id,
  c.name AS company_name,
  r.id AS role_id,
  r.code AS role_code,
  r.name AS role_name
FROM users u
JOIN company_users cu ON cu.user_id = u.id
JOIN companies c ON c.id = cu.company_id
JOIN roles r ON r.id = cu.role_id
WHERE u.username = '{{InputUsername.text}}'
  AND u.password_hash = '{{InputPassword.text}}'
  AND u.is_active = 1
LIMIT 1;
