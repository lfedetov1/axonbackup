SELECT
    u.id AS userId,
    u.username AS username,
    cu.company_id AS companyId,
    w.id AS warehouseId,
    w.code AS warehouseCode,
    w.name AS warehouseName
FROM users u
JOIN company_users cu ON cu.user_id = u.id
LEFT JOIN warehouses w ON w.company_id = cu.company_id
WHERE u.username = {{InputUsername.text}}
  AND u.is_active = 1
LIMIT 1;
