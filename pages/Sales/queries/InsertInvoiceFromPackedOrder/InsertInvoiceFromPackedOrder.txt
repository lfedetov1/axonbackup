INSERT INTO documents (
    company_id,
    warehouse_id,
    partner_id,
    created_by_user_id,
    document_type,
    status,
    posting_status,
    document_number,
    document_date,
    due_date,
    currency_code,
    exchange_rate,
    source_document_id,
    note,
    subtotal_amount,
    tax_amount,
    discount_amount,
    total_amount,
    paid_amount,
    sales_channel,
    payment_method,
    customer_reference,
    internal_reference
)
SELECT
    d.company_id,
    d.warehouse_id,
    d.partner_id,
    {{appsmith.store.userId}},
    'SALES_INVOICE',
    'DRAFT',
    'NOT_POSTED',
    CONCAT('INV-PACK-', d.id),
    CURDATE(),
    d.due_date,
    d.currency_code,
    d.exchange_rate,
    d.id,
    CONCAT(COALESCE(d.note, ''), '\nCreated from packing'),
    d.subtotal_amount,
    d.tax_amount,
    d.discount_amount,
    d.total_amount,
    0,
    d.sales_channel,
    d.payment_method,
    d.customer_reference,
    d.internal_reference
FROM documents d
WHERE d.id = {{this.params.orderId}};

SELECT LAST_INSERT_ID() AS invoiceId;
