SELECT
    d.id AS `ID`,
    d.document_number AS `Number`,
    d.document_type AS `Document Type`,
    DATE_FORMAT(d.document_date, '%d.%m.%Y') AS `Date`,
    bp.code AS `Customer Code`,
    bp.name AS `Customer`,
    w.name AS `Warehouse`,
    d.payment_method AS `Payment Method`,
    d.status AS `Status`,
    d.posting_status AS `Posting Status`,

    CASE
        WHEN d.document_type = 'CREDIT_NOTE'
        THEN -ABS(COALESCE(d.total_amount, 0))
        ELSE COALESCE(d.total_amount, 0)
    END AS `Total`,

    CASE
        WHEN d.document_type = 'CREDIT_NOTE'
        THEN -ABS(COALESCE(d.paid_amount, 0))
        ELSE COALESCE(d.paid_amount, 0)
    END AS `Paid`,

    DATE_FORMAT(d.created_at, '%d.%m.%Y %H:%i:%s') AS `Created At`
FROM documents d
LEFT JOIN business_partners bp ON bp.id = d.partner_id
LEFT JOIN warehouses w ON w.id = d.warehouse_id
WHERE d.document_type IN ('SALES_INVOICE', 'CREDIT_NOTE')
  AND d.document_date >= STR_TO_DATE('{{ moment(dtpFrom.selectedDate).format("DD.MM.YYYY") }}', '%d.%m.%Y')
  AND d.document_date < DATE_ADD(
      STR_TO_DATE('{{ moment(dtpTo.selectedDate).format("DD.MM.YYYY") }}', '%d.%m.%Y'),
      INTERVAL 1 DAY
  )
  AND (
      {{ Number(appsmith.store.warehouseId1 || 0) }} = 0
      OR d.warehouse_id = {{ Number(appsmith.store.warehouseId1 || 0) }}
      OR IFNULL(d.warehouse_id, 0) = 0
  )
ORDER BY d.document_date DESC, d.id DESC;
