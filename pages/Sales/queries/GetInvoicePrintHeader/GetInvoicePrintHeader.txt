SELECT
  si.ID AS id,
  si.CompanyID AS companyId,
  c.logo_path AS company_logo_path,
  si.Number AS documentNumber,
  si.Date AS documentDate,
  si.DueDate AS dueDate,
  si.PartnerID AS partnerId,
  si.CustomerCode AS partnerCode,
  si.Customer AS partnerName,
  bp.legal_name AS partnerLegalName,
  bp.address_line1 AS partnerAddress,
  bp.postal_code AS partnerPostalCode,
  bp.city_name AS partnerCity,
  bp.tax_number AS partnerTaxNumber,
  bp.email AS partnerEmail,
  bp.phone AS partnerPhone,
  si.WarehouseID AS warehouseId,
  si.Warehouse AS warehouseName,
  si.PaymentMethod AS paymentMethod,
  si.Subtotal AS subtotalAmount,
  si.Tax AS taxAmount,
  si.Discount AS discountAmount,
  si.Total AS totalAmount,
  si.Paid AS paidAmount,
  si.OpenAmount AS openAmount,
  si.CreatedBy AS createdBy,
  si.CreatedAt AS createdAt,
  c.name AS companyName,
  c.address_line1 AS companyAddress,
  c.postal_code AS companyPostalCode,
  c.city AS companyCity,
  c.tax_number AS companyTaxNumber,
  c.email AS companyEmail,
  c.phone AS companyPhone,
  c.base_currency_code AS currencyCode,
  'SALES_INVOICE' AS documentType
FROM SalesInvoices si
JOIN companies c ON c.id = si.CompanyID
LEFT JOIN business_partners bp ON bp.id = si.PartnerID
WHERE si.Number = {{ appsmith.store.selectedDocumentId }};
