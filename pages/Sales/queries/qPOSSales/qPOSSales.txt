SELECT
    ID AS `ID`,
    Number AS `Number`,
    Date AS `Date`,
    CustomerCode AS `Customer Code`,
    Customer AS `Customer`,
    Warehouse AS `Warehouse`,
    PaymentMethod AS `Payment Method`,
    Cashier AS `Cashier`,
    Status AS `Status`,
    PostingStatus AS `Posting Status`,
    Total AS `Total`,
    Paid AS `Paid`,
    DATE_FORMAT(CreatedAt, '%d.%m.%Y %H:%i:%s') AS `Created At`
FROM POSSales
WHERE Date >= '{{ moment(dtpFrom.selectedDate).format("YYYY-MM-DD") }}'
  AND Date < DATE_ADD('{{ moment(dtpTo.selectedDate).format("YYYY-MM-DD") }}', INTERVAL 1 DAY)
  AND (
      {{appsmith.store.warehouseId || 0}} = 0
      OR WarehouseID = {{appsmith.store.warehouseId || 0}}
  )
ORDER BY Date DESC, ID DESC;
