SELECT CONCAT(
  'INV-',
  LPAD(
    IFNULL(
      MAX(
        CAST(
          SUBSTRING_INDEX(SUBSTRING_INDEX(document_number, '-', 2), '-', -1)
          AS UNSIGNED
        )
      ),
      0
    ) + 1,
    3,
    '0'
  ),
  '-MAIN'
) AS invoiceNumber
FROM documents
WHERE company_id = {{ appsmith.store.companyId }}
  AND document_type = 'SALES_INVOICE'
  AND document_number LIKE 'INV-%-MAIN';
