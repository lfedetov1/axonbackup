SELECT
    d.id AS documentId,
    d.document_number AS documentNumber,
    d.document_date AS documentDate,
    d.created_at AS createdAt,
    d.payment_method AS paymentMethod,
    d.card_type AS cardType,
    d.currency_code AS currencyCode,
    d.subtotal_amount AS subtotalAmount,
    d.discount_amount AS discountAmount,
    d.tax_amount AS taxAmount,
    d.total_amount AS totalAmount,
    d.paid_amount AS paidAmount,
    (d.total_amount - d.paid_amount) AS openAmount,
    d.status AS status,
    d.posting_status AS postingStatus,

    d.zki AS zki,
    d.jir AS jir,
    d.fiscal_verification_url AS fiscalVerificationUrl,

    c.name AS companyName,
    c.tax_number AS companyTaxNumber,
    c.registration_number AS companyRegistrationNumber,
    c.email AS companyEmail,
    c.phone AS companyPhone,
    c.address_line1 AS companyAddress,
    c.city AS companyCity,
    c.postal_code AS companyPostalCode,
    c.logo_path AS company_logo_path,

    w.name AS warehouseName,
    w.code AS warehouseCode,

    u.username AS createdBy,
    cu.username AS cashier
FROM documents d
JOIN companies c ON c.id = d.company_id
LEFT JOIN warehouses w ON w.id = d.warehouse_id
LEFT JOIN users u ON u.id = d.created_by_user_id
LEFT JOIN users cu ON cu.id = d.cashier_user_id
WHERE d.document_number = '{{ tblPOSSales.selectedRow ? tblPOSSales.selectedRow.Number : "" }}'
  AND d.document_type = 'POS_SALE'
LIMIT 1;
