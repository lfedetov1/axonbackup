SELECT
  COALESCE(today_revenue.today_revenue, 0) AS today_revenue,
  COALESCE(today_revenue.today_paid, 0) AS today_paid,
  COALESCE(today_revenue.today_document_count, 0) AS today_document_count,

  COALESCE(open_invoices.open_invoice_count, 0) AS open_invoice_count,
  COALESCE(open_invoices.open_invoice_amount, 0) AS open_invoice_amount,
  COALESCE(open_invoices.overdue_invoice_amount, 0) AS overdue_invoice_amount,

  COALESCE(low_stock.low_stock_count, 0) AS low_stock_count,
  COALESCE(low_stock.estimated_restock_value, 0) AS estimated_restock_value,

  COALESCE(alerts.active_alert_count, 0) AS active_alert_count

FROM
  (
    SELECT
      SUM(d.total_amount) AS today_revenue,
      SUM(d.paid_amount) AS today_paid,
      COUNT(d.id) AS today_document_count
    FROM documents d
    WHERE d.company_id = {{ appsmith.store.companyId }}
      AND (
        {{ appsmith.store.warehouseId || 0 }} = 0
        OR d.warehouse_id = {{ appsmith.store.warehouseId || 0 }}
      )
      AND d.document_type IN ('SALES_INVOICE', 'POS_SALE')
      AND d.status <> 'CANCELLED'
      AND d.document_date = CURDATE()
  ) today_revenue

CROSS JOIN
  (
    SELECT
      COUNT(d.id) AS open_invoice_count,
      SUM(d.total_amount - d.paid_amount) AS open_invoice_amount,
      SUM(CASE WHEN d.due_date < CURDATE() THEN d.total_amount - d.paid_amount ELSE 0 END) AS overdue_invoice_amount
    FROM documents d
    WHERE d.company_id = {{ appsmith.store.companyId }}
      AND (
        {{ appsmith.store.warehouseId || 0 }} = 0
        OR d.warehouse_id = {{ appsmith.store.warehouseId || 0 }}
      )
      AND d.document_type = 'SALES_INVOICE'
      AND d.status <> 'CANCELLED'
      AND d.total_amount > d.paid_amount
  ) open_invoices

CROSS JOIN
  (
    SELECT
      COUNT(*) AS low_stock_count,
      SUM(GREATEST(p.minimum_stock - COALESCE(ib.quantity_balance, 0), 0) * p.purchase_price) AS estimated_restock_value
    FROM products p
    JOIN warehouses w
      ON w.company_id = p.company_id
    LEFT JOIN v_inventory_balance ib
      ON ib.product_id = p.id
     AND ib.warehouse_id = w.id
     AND ib.company_id = p.company_id
    WHERE p.company_id = {{ appsmith.store.companyId }}
      AND (
        {{ appsmith.store.warehouseId || 0 }} = 0
        OR w.id = {{ appsmith.store.warehouseId || 0 }}
      )
      AND p.is_active = 1
      AND p.track_stock = 1
      AND w.is_active = 1
      AND COALESCE(ib.quantity_balance, 0) < p.minimum_stock
  ) low_stock

CROSS JOIN
  (
    SELECT COUNT(*) AS active_alert_count
    FROM (
      SELECT 1
      FROM documents
      WHERE company_id = {{ appsmith.store.companyId }}
        AND (
          {{ appsmith.store.warehouseId || 0 }} = 0
          OR warehouse_id = {{ appsmith.store.warehouseId || 0 }}
        )
        AND document_type IN ('SALES_INVOICE', 'POS_SALE', 'DELIVERY_NOTE', 'SALES_RETURN')
        AND status = 'CONFIRMED'
        AND posting_status IN ('NOT_POSTED', 'READY')
      HAVING COUNT(*) > 0

      UNION ALL

      SELECT 1
      FROM documents
      WHERE company_id = {{ appsmith.store.companyId }}
        AND (
          {{ appsmith.store.warehouseId || 0 }} = 0
          OR warehouse_id = {{ appsmith.store.warehouseId || 0 }}
        )
        AND document_type = 'SALES_INVOICE'
        AND status <> 'CANCELLED'
        AND due_date < CURDATE()
        AND total_amount > paid_amount
      HAVING COUNT(*) > 0

      UNION ALL

      SELECT 1
      FROM documents
      WHERE company_id = {{ appsmith.store.companyId }}
        AND (
          {{ appsmith.store.warehouseId || 0 }} = 0
          OR warehouse_id = {{ appsmith.store.warehouseId || 0 }}
        )
        AND document_type = 'QUOTE'
        AND status = 'DRAFT'
        AND valid_until < CURDATE()
      HAVING COUNT(*) > 0

      UNION ALL

      SELECT 1
      FROM cash_registers cr
      WHERE cr.company_id = {{ appsmith.store.companyId }}
        AND (
          {{ appsmith.store.warehouseId || 0 }} = 0
          OR cr.warehouse_id = {{ appsmith.store.warehouseId || 0 }}
        )
        AND cr.is_active = 1
        AND NOT EXISTS (
          SELECT 1
          FROM cash_register_closings crc
          WHERE crc.cash_register_id = cr.id
            AND crc.closing_date = DATE_SUB(CURDATE(), INTERVAL 1 DAY)
            AND crc.status = 'CLOSED'
        )
      HAVING COUNT(*) > 0
    ) active_alerts
  ) alerts;
